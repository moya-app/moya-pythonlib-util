15class RateLimitSettings(MoyaSettings): 

16 """ 

17 Extract environment variable APP_RATELIMITS and use it to configure 

18 ratelimiters. 

19 

20 A default entry of "*" can be used to set a default rate limit for all 

21 endpoints which do not have a specific limiter config set up. 

22 """ 

23 

24 ratelimits: dict[str, RateLimit] 

25 

26 def get(self, endpoint: str, default: RateLimit | None = None) -> RateLimit: 

27 """ 

28 Get the ratelimiter for the given endpoint name, if exists 

29 """ 

30 if endpoint in self.ratelimits: 

31 return self.ratelimits[endpoint] 

32 elif "*" in self.ratelimits: 

33 return self.ratelimits["*"] 

34 elif default is not None: 

35 return default 

36 else: 

37 raise KeyError( 

38 f"No ratelimits configured for limiter '{endpoint}'." 

39 " Ensure that APP_RATELIMITS contains at least an entry for '*'," 

40 " or that default_limits is set in the config" 

41 )

51class _RateLimiter: 

52 def __init__( 

53 self, 

54 limiter: LimitBase, 

55 user_id_decorator: t.TypeAlias = t.Annotated[None, Depends(_null_fn)], 

56 ) -> None: 

57 """ 

58 Create a new rate limiter with the given rates. This class is intended 

59 to wrap details about different limiters and allow for easy integration 

60 into FastAPI. 

61 

62 :param limiter: The limiter to use. This should be an instance of LimitBase. 

63 :param user_id_decorator: The decorator to use to get the user_id from 

64 the request. This should be an t.Annotated function (probably using 

65 Depends()) returning a string or None. If the function returns None 

66 then no ratelimits will be applied to this user (which is 

67 potentially useful for admin users to bypass ratelimits). 

68 

69 """ 

70 self.limiter = limiter 

71 self.user_id_decorator = user_id_decorator 

72 

73 async def reset_user(self, user_id: str) -> None: 

74 """ 

75 Clear all ratelimits for this limiter for the specified user 

76 """ 

77 await self.limiter.flush_user(user_id) 

78 

79 async def reset_all(self) -> None: 

80 """ 

81 Reset all ratelimits for all users using this limiter 

82 """ 

83 await self.limiter.reset()

85 @property 

86 def dependency(self) -> t.Any: # Any is the type of Depends() anyway 

87 if self.limiter.rates.is_empty: 

88 return Depends(_null_fn) 

89 

90 async def fn(user_id: self.user_id_decorator) -> None: # type: ignore 

91 # TODO: Could use dynamic rate limits here on a per-path basis by 

92 # using something like the following as the key prefix: 

93 # [request.method, request['route'].path] 

94 

95 if user_id is None: 

96 return 

97 

98 try: 

99 await self.limiter.try_ratelimit(user_id) 

100 except RateLimitExceeded: 

101 raise HTTPException(status_code=status.HTTP_429_TOO_MANY_REQUESTS, detail="Too many requests") 

102 

103 return Depends(fn)

109def RateLimiter( 

110 name: str, 

111 user_id_decorator: t.Any, # Must be a Depends() function which returns a str or None 

112 default_limits: t.Optional[RateLimit] = None, 

113 limiter_class: t.Type[LimitBase] = RedisLimiter, 

114) -> _RateLimiter: 

115 """ 

116 Easy fastapi dependency to create a rate limiter from environment vars. 

117 

118 :param name: The name of the limiter to use. This should be the same as the 

119 name of the endpoint you want to limit and will have limiter config 

120 pulled from that item in the APP_RATELIMITS config variable. 

121 :param user_id_decorator: The decorator to use to get the user_id from the 

122 request. This must be specified and should be a Depends() function 

123 returning a str or None for no ratelimit for this particular user. 

124 :param default_limits: The default limits to use if no limits are specified 

125 in the environment, otherwise if no default limits are specified in the 

126 environment a startup-time error will be raised. 

127 :param limiter_class: The class to use for the limiter. This should be a 

128 subclass of LimitBase. This defaults to RedisLimiter, but could be set 

129 to MemLimiter for testing. 

130 

131 Usage: 

132 

133 async def ensure_verified( 

134 credentials: t.Annotated[HTTPBasicCredentials, Depends(HTTPBasic())] 

135 ): 

136 return credentials.username 

137 

138 limiter = RateLimiter("foo", Depends(ensure_verified)) 

139 @app.get("/foo", dependencies=[limiter.dependency]) 

140 async def foo(): 

141 await limiter.reset_user("foo@bar.com") 

142 

143 and set APP_RATELIMITS='{"foo": {"per_minute": 10, "per_day": 20}}' in your environment 

144 """ 

145 

146 settings = RateLimitSettings() 

147 user_id_decorator = user_id_decorator 

148 if t.get_origin(user_id_decorator) is not t.Annotated: 

149 user_id_decorator = t.Annotated[str, user_id_decorator] 

150 

151 ratelimiter = _RateLimiter(limiter_class(settings.get(name, default_limits), name), user_id_decorator=user_id_decorator) 

152 # TODO: I know this is hacky as it holds all references ever created. But with FastAPI this is probably 

153 # reasonable to assume. 

154 _all_fastapi_ratelimiters.append(ratelimiter) 

155 return ratelimiter
